    2 Concurrent

    Concurrent

    The ability of different parts or units of a program, algorithm or problem to be executed out-of-order or in partial order, without affecting the final outcome.

    1 еӨҡеӨ„зҗҶеҷЁзј–зЁӢ

    1.1 зәҝзЁӢпјҡе№¶еҸ‘зҡ„еҹәжң¬еҚ•дҪҚ

    еӨҡдёӘзәҝзЁӢжү§иЎҢжөҒе…ұдә«еҶ…еӯҳпјҡ

    • жү§иЎҢжөҒ жӢҘжңү зӢ¬з«Ӣзҡ„ Stack & Regs
    • е…ұдә«е…ЁйғЁзҡ„еҶ…еӯҳпјҲжҢҮй’ҲеҸҜд»Ҙзӣёдә’еј•з”Ёпјү

    thread.hпјҡз®ҖеҢ–зәҝзЁӢ API

    еҹәдәҺ POSIX Threads е®һзҺ°

    • create(fn) еҲӣе»әд»Ҙ fn дёәе…ҘеҸЈеҮҪж•°зҡ„зәҝзЁӢпјҢе№¶ з«ӢеҚі ејҖе§Ӣжү§иЎҢ
      • void fn(int tid){} е…¶дёӯ tid д»Һ 1 ејҖе§Ӣзј–еҸ·
    • join() зӯүеҫ…жүҖжңүиҝҗиЎҢзәҝзЁӢзҡ„ fn иҝ”еӣһ
      • main() иҝ”еӣһж—¶з»“жқҹжүҖжңүзӯүеҫ…иҝӣзЁӢ
    #include "thread.h"

void Ta() { while(1) { printf("a");} }
void Tb() { while(2) { printf("b");} }

int main() {
    create(Ta);
    create(Tb);
}

    1.2 еҺҹеӯҗжҖ§

    еҺҹеӯҗжҖ§

    • дёҖж®өд»Јз Ғ жү§иЎҢж—¶ зӢ¬еҚ  ж•ҙдёӘи®Ўз®—жңәзі»з»ҹ
    • иҝҷдёҖеҹәжң¬еҒҮи®ҫеңЁзҺ°д»Јзҡ„еӨҡеӨ„зҗҶеҷЁзі»з»ҹдёҠдёҚеҶҚжҲҗз«Ӣ
      • еҚ•еӨ„зҗҶеҷЁеӨҡзәҝзЁӢпјҡеҚ•дёҖзәҝзЁӢеңЁиҝҗиЎҢж—¶еҸҜиғҪиў«дёӯж–ӯ
      • еӨҡеӨ„зҗҶеҷЁеӨҡзәҝзЁӢпјҡеӨҡдёӘеӨ„зҗҶеҷЁе№¶иЎҢжү§иЎҢ

    дёәжӯӨпјҢжҲ‘д»¬йңҖиҰҒе®һзҺ° lock() & unlock() е®һзҺ° дёҙз•ҢеҢә(critical section) еҶ…д»Јз Ғзҡ„ з»қеҜ№дёІиЎҢеҢ–

    зЁӢеәҸзҡ„е…¶дҪҷйғЁеҲҶд»ҚеҸҜд»Ҙ е№¶иЎҢжү§иЎҢ

    • 99% зҡ„е№¶еҸ‘й—®йўҳйғҪеҸҜд»ҘйҖҡиҝҮ йҳҹеҲ— и§ЈеҶіпјҡ
      • е°ҶеӨ§й—®йўҳдё”еҲҶдёәиӢҘе№ІеҸҜд»Ҙ е№¶иЎҢ зҡ„е°Ҹд»»еҠЎ
      • дҪҝз”Ё worker thread д»Һиў« lock дҝқжҠӨзҡ„йҳҹеҲ—дёӯйўҶеҸ–д»»еҠЎ
      • йҷӨз»қеҜ№дёІиЎҢйғЁеҲҶпјҢе…¶дҪҷйғЁеҲҶеқҮеҸҜиҺ·еҫ—зәҝжҖ§еҠ йҖҹ

    1.3 йЎәеәҸ

    Compiler еҜ№еҶ…еӯҳи®ҝй—®йҮҮз”ЁдәҶ Eventually Consistent зҡ„еӨ„зҗҶзӯ–з•Ҙ

    • й—®йўҳпјҡ иҝҷеҜјиҮҙйҖҡиҝҮ е…ұдә«еҶ…еӯҳ е®һзҺ°зҡ„зәҝзЁӢеҗҢжӯҘеӨұж•Ҳ
    • и§ЈеҶіпјҡеҗ‘д»Јз ҒдёӯжҸ’е…Ҙ дјҳеҢ–дёҚеҸҜз©ҝи¶Ҡ зҡ„ barrier

while(!done);

    1.4 еҸҜи§ҒжҖ§

    barrier дҝқиҜҒиөӢеҖјиў«з«ӢеҲ»еҶҷе…ҘеҶ…еӯҳ
    int x = 0, y = 0; 
void T1() { 
    x = 1; 
    asm volatile("" : : "memory"); // compiler barrier 
    printf("y = %d\n", y); 
} 

void T2() { 
    y = 1; 
    asm volatile("" : : "memory"); // compiler barrier 
    printf("x = %d\n", x); 
}

    еҸҚеӨҚи°ғз”Ё T1 & T2пјҢжҲ‘д»¬зңӢеҲ°зҡ„иҫ“еҮәеҸӘжңү 00 01 10

    => иҝҷжҳҜеӣ дёә еӨ„зҗҶеҷЁ жң¬иә«д№ҹжҳҜпјҲеҠЁжҖҒзҡ„пјүзј–иҜ‘еҷЁ

    • еҚ•дёӘеӨ„зҗҶеҷЁдҪҝз”Ёз”өи·Ҝе°Ҷ Assembly вҖңзј–иҜ‘вҖқ жҲҗжӣҙе°Ҹзҡ„ \(\mu ops\) жҜҸдёӘ \(\mu ops\) йғҪжңү Fetch - Issue - Execute - Commit еӣӣдёӘйҳ¶ж®ө
    • еңЁд»»ж„Ҹж—¶ж®өпјҢеӨ„зҗҶеҷЁйғҪз»ҙжҠӨдәҶдёҖдёӘ \(\mu ops\) pool
      • жҜҸдёҖе‘ЁжңҹеҶ…пјҢеҗ‘ pool иЎҘе…… е°ҪеҸҜиғҪеӨҡзҡ„ \(\mu ops\)пјҲеӨҡеҸ‘е°„пјү
      • пјҲдёҚиҝқеҸҚзј–иҜ‘жӯЈзЎ®жҖ§зҡ„еүҚжҸҗдёӢпјүжҜҸдёҖе‘Ёжңҹ жү§иЎҢ е°ҪеҸҜиғҪеӨҡзҡ„ \(\mu ops\)пјҲд№ұеәҸжү§иЎҢгҖҒжҢүеәҸжҸҗдәӨпјү

    еҚ•еӨ„зҗҶеҷЁдёҠж»Ўи¶і eventual memory consistency зҡ„жү§иЎҢйЎәеәҸпјҢеҸҜиғҪж— жі•еңЁеӨҡеӨ„зҗҶеҷЁдёҠеәҸеҲ—еҢ–

    еңЁ write(x) еҸ‘з”ҹ cache miss ж—¶пјҢread(y) дјҡе…Ҳиў«жү§иЎҢпјҲжңҖеӨ§еҢ–еӨ„зҗҶеҷЁжҖ§иғҪпјү

    => еҸҜиғҪеҜјиҮҙдёӨдёӘиҜ»ж“ҚдҪңеқҮе…ҲдәҺеҶҷж“ҚдҪңжү§иЎҢпјҢеҜјиҮҙиҫ“еҮә 00

    йЎәеәҸдёҖиҮҙжҖ§зҡ„е®һзҺ°

    1. Memory Barrier дҪҝз”Ё compiler barrier + mfence жҢҮд»ӨпјҢејәиҝ«е…ҲеҶҷеҗҺиҜ»
    2. жҸ’е…ҘеҺҹеӯҗжҢҮд»ӨпјҲlock prefix, lr/sc ...пјү

    2 е№¶еҸ‘жҺ§еҲ¶

    2.1 дә’ж–Ҙ Mutual Exclusion

    !!! info вҖң'дә’ж–Ҙ' иЎЁзӨәдёӨдёӘ зәҝзЁӢ дёҚиғҪеҗҢж—¶жү§иЎҢвҖқ

    еҰӮжһңжҲ‘д»¬еёҢжңӣйҖҡиҝҮ е…ұдә«еҶ…еӯҳ зҡ„ж–№ејҸдҝқиҜҒдёӨдёӘзәҝзЁӢеңЁзү№е®ҡеҢәж®ө вҖңдә’ж–ҘвҖқпјҢйӮЈд№ҲжҲ‘д»¬е°ұеә”еҪ“дҪҝз”Ё lock() & unlock() ж“ҚдҪңеҜ№е…¶иҝӣиЎҢеҢ…иЈ№гҖӮдёӢйқўжҳҜдёҖдёӘ еӨұиҙҘ зҡ„дҫӢеӯҗпјҡ

    bool locked = false; // еҲқе§Ӣжғ…еҶөжңӘдёҠй”Ғ

void critical_section() {
    // 1. test
    while(locked) {} // дёҠй”Ғж—¶иҮӘз’Ү
    // 2. lock
    locked = true;

    // critical section (дә’ж–ҘзүҮж®ө)

    // 3. UNlock
    locked = false;
}

    з”ұдәҺж— жі•дҝқиҜҒеҜ№ locked еҸҳйҮҸ load & store ж“ҚдҪңзҡ„еҺҹеӯҗжҖ§пјҢиҝҷдёҖе®һзҺ°жҳҜй”ҷиҜҜзҡ„

    жүҖд»ҘпјҢе®һзҺ°дә’ж–Ҙзҡ„ ж №жң¬еӣ°йҡҫ еңЁдәҺпјҡжҲ‘д»¬дёҚиғҪ еҗҢж—¶ иҜ» / еҶҷ е…ұдә«еҶ…еӯҳпјҒ

    • load ж—¶дёҚиғҪеҶҷпјҡзңӢеҲ°зҡ„еҶ…е®№еҸҜиғҪй©¬дёҠе°ұиҝҮжңҹдәҶ
    • store ж—¶дёҚиғҪиҜ»пјҡдҝ®ж”№зҡ„еҶ…е®№еҸҜиғҪиў«еҲ«дәәиҰҶзӣ–

    2.1.1 Peterson з®—жі•

    иәІиҝӣеҺ•жүҖй”ҒдёҠй—ЁпјҢжҲ‘е°ұжҠҠе…Ёдё–з•Ңдәәй”ҒеңЁдәҶеҺ•жүҖеӨ–

    A е’Ң B дәүз”ЁеҺ•жүҖзҡ„еҢ…еҺў

    1. жғі иҝӣе…ҘеҢ…еҺўд№ӢеүҚпјҢA/B йғҪиҰҒе…Ҳдёҫиө·иҮӘе·ұзҡ„ж——еӯҗ

      • A зЎ®и®Өж——еӯҗдёҫеҘҪд»ҘеҗҺпјҢеҫҖеҺ•жүҖй—ЁдёҠиҙҙдёҠ вҖңB жӯЈеңЁдҪҝз”ЁвҖқ зҡ„ж Үзӯҫ
      • B зЎ®и®Өж——еӯҗдёҫеҘҪд»ҘеҗҺпјҢеҫҖеҺ•жүҖй—ЁдёҠиҙҙдёҠ вҖңA жӯЈеңЁдҪҝз”ЁвҖқ зҡ„ж Үзӯҫ

    2. еҰӮжһңеҜ№ж–№зҡ„ж——еӯҗдёҫиө·жқҘпјҢдё”й—ЁдёҠзҡ„еҗҚеӯ—дёҚжҳҜиҮӘе·ұпјҢзӯүеҫ…пјӣеҗҰеҲҷиҝӣе…ҘеҢ…еҺў

      еӣ дёәжүӢеҝ«зҡ„дәәеҶҷзҡ„ turn дјҡиў«жүӢж…ўзҡ„иҰҶзӣ–пјҲ turn = жүӢеҝ«дәәзҡ„ Labelпјү

    3. еҮәеҢ…еҺўеҗҺпјҢж”ҫдёӢиҮӘе·ұзҡ„ж——еӯҗ

    bool flagA = false, flagB = false;
char turn = 'A';

void A() {
    while(1) {
        flagA = true; // дёҫиө· иҮӘе·ұзҡ„ ж——еӯҗ
        turn = 'B';   // иҙҙ   еҜ№йқўзҡ„ ж Үзӯҫ

        while(flagB && turn == 'B') {};

        // critical section

        flagA = false; // ж”ҫдёӢиҮӘе·ұзҡ„ж——еӯҗ
    }
}

void B() {
    while(1) {
        flagB = true; // дёҫиө· иҮӘе·ұзҡ„ ж——еӯҗ
        turn = 'A';   // иҙҙ   еҜ№йқўзҡ„ ж Үзӯҫ

        while(flagA && turn == 'A') {};

        // critical section

        flagB = false; // ж”ҫдёӢиҮӘе·ұзҡ„ж——еӯҗ
    }
}

void Producer() {
    while(1) {
        mutex_lock(&mutex); // иҺ·еҸ–дә’ж–Ҙй”Ғ
        while(!(count != n)) {
            cond_wait(&cv, &lk); // е·ҰжӢ¬еҸ·ж»Ў => йҮҠж”ҫ lkпјҢиҮӘиә«йҳ»еЎһ
        }
        // дёҙз•ҢеҢә
        assert(cond != n); // е·ҰжӢ¬еҸ·жңӘж»ЎпјҢmutex зЎ®дҝқжңҹй—ҙеҶ… cond дёҚеҸҳ
        printf("(");
        // йҖҖеҮәдёҙз•ҢеҢә
        cond ++;
        cond_broadcast(&cv); // е”ӨйҶ’жүҖжңү consumer
        mutex_unlock(&lk);
    }
}

void Consumer() {
    while(1) {
        mutex_lock(&lk);
        while(!(cond != 0)) {
            pthread_cond_wait(&cv, &lk); // ж— е·ҰжӢ¬еҸ·
        }
        // дёҙз•ҢеҢә
        assert(count != 0);
        printf(")");
        // йҖҖеҮәдёҙз•ҢеҢә
        cond --;
        cond_broadcast(&cv);
        mutex_unlock(&lk);
    }
}
