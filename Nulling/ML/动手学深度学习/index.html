    AutoGrad

    自动梯度计算

    Sample：求函数 \(y = 2x^Tx\) 关于列向量 \(x\) 的梯度

    from mxnet import autograd, nd

# 初始化变量 x
x = nd.arrange(4).reshape((4,1))

# 申请用于存储梯度的内存
x.attach_grad()

# 调用 record 函数记录与梯度相关的计算（默认不记录以减少开销）
with autograd.record():
    y = 2* nd.dot(x.T, x)

# 因为 y = (4,1) * (1,4) = (1,1) 为标量，可以直接用 backward() 求梯度
# 若 y 并非标量，则会先调用 y.sum()
y.backward()

# 验证 y ?= 4x
assert (x.grad - 4*x).norm().asscalar() == 0

# 输出梯度计算结果
print(x.grad)

    即便函数计算中包含了 while 和 if 控制语句，我们也可能对变量求梯度：

    • 函数 f(a) 的输出取决于 a 的值

      def f(a):
    b = a * 2
    while b.norm().asscalar() < 1000:
        b = b * 2
    if b.sum().asscalar() > 0:
        c = b
    else:
        c = b * 100
    return c

    • 我们同样可以使用 backward() 求梯度

      a = nd.random.normal(shape=1)
a.attach_grad()
with autograd.record():
    c = f(a)
c.backward()

# 由于 f(a) = a * x -> f(a)' = x = c/a
assert a.grad == c / a

    Linera Regression 线性回归

    使用一条直线来拟合训练数据，并使用参数 \((w,b)\) 进行预测：

    \[ \hat{y} = wX + b \]

    优化目标：找到一组 \((w,b)\) 以最小化平方误差项：

    \[ \sum(\hat{y}_i - y_i)^2 \]

    我们可以将 线性回归 模型视为一个使用 \(g(x) = x\) 作为激活函数的单层神经网络

    1 生成数据集

    我们根据以下公式生成训练用的数据集：

    \[ y[i] = 2 * x[i][0] - 3.4*x[i][1] + 4.2 + \text{noise} \]
    • 噪声符合 \(\mu=0,\sigma=0.1\) 的正态分布
    • 理想情况下参数 \((w,b) = ([2,-3.4]^T, 4.2)\)
    from mxnet import ndarray as nd

n_feature  = 2    # 每个样本含有 2 个 feature
n_example = 1000 # 生成 m=1000 的数据集

# 理想情况下的参数
true_w = [2, -3.4]
true_b = 4.2

# 生成 2x1000 个 features
x = nd.random_nromal(shape=(n_examples, n_feature))

# 按公式生成 label
y = true_w[0] * x[:, 0] + true_w[1] * x[:, 1] + true_b
y += 0.01 * nd.random_normal(shape=y.shape) # 添加正态分布的噪声

    2 小批量读取数据

    封装以下函数，每次返回 batch_size 个 随机样本

    # features 对应训练集中的 X
# labels   对应训练集中的 Y
def data_iter(batch_size, features, labels):
    n_example = len(features)
    indices = list(range(n_example))
    # 读取顺序是随机的
    random.shuffle(indicies)
    for i in range(0, n_example, batch_size):
        j = nd.array(indices[i: min(i+btach_size, n_example)])
        yield features.take(j), labels.take(j) # 抽取下标 j 对应的样本

    3 定义并训练模型

    from mxnet import autograd

# 初始化参数 w,b
w = nd.random.normal(scale=0.01, shape=(n_feature, 1))
b = nd.zeros(shape=(1,))

# 由于之后需要对 w,b 进行梯度下降，此处需要创建对应的梯度
w.attach_grad()
b.attach_grad()

# 定义线性回归模型（预测逻辑）
def linreg(X, w, b):
    return nd.dot(X, w) + b

# 定义损失函数（均方误差）
def squared_loss(y_hat, y):
    # 统一两者的 shape 避免自动广播
    return (y_hat - y.reshape(y_hat.shape)) ** 2

# 随机梯度下降（参数更新）
def SGD(params, alpha, batch_size):
    for p in params:
        p[:] = p - alpha * p.grad / batch_size

# 训练模型
epochs = 5
alpha = 0.001
for e in range(epochs):
    total_loss = 0
    for data, label in data_iter(batch_size, x, y): # 每次对一个 batch 进行梯度下降
        # 前向传导计算误差
        with autograd.record():
            output = linreg(data) # 使用当前参数进行预测
            loss = square_loss(output, label)
        # 反向传播进行梯度下降
        loss.backward()
        SGD(params, alpha, batch_size)
        # 计算总误差
        total_los += nd.sum(loss).asscalar()
    print("in Epoch[%d], average loss = %f" % (e, total_loss/n_example))

    L2 范数正则化

    对 绝对值较大 的参数进行惩罚，需要额外定义正则化系数 \(\lambda\)

    L2 范数正则化项定义如下（此时 Cost = Loss + L2）：

    \[ L_2 = \lambda \cdot \sum_{p \in params} \|p\|^2 \]

    在 测试模型时 需要令 \(\lambda = 0\)

    正则化的 Gulon 实现
    model = nn.Sequential()
model.add(nn.Dense(1))
model.initialize(init.Normal(sigma=1))

# 对 trainer 添加超参数 wd(weight decade) 即可
# 对参数 W 进行正则化
trainer_w = gluon.Trainer(
    model.collect_params('.*weight'), 'sgd',
    {'learning_rate': 0.01, 'wd': wd}
)
# 不对参数 b 进行正则化
trainer_b = gluon.Trainer(
    model.collect_params('.*bias'), 'sgd',
    {'learning_rate': 0.01}
)

# 此时，每一轮训练应该对 w,b 分别调用 step() 以同时更新
with autograd.record():
    output = model(data)
    loss   = loss(output, label)

loss.backward()
trainer_w.step(batch_size)
trainer_b.step(batch_size)

    使用 nn.Block 定义模型

    nn.Sequential 其实是 nn.Block 的简化形式，我们可以通过继承基类 nn.Block 来定义模型。你至少需要实现以下两个方法：

    • __init__ 构造函数，用于初始化参数
    • forward 用于定义前向传播逻辑
    重现实现多层感知机
    class MLP(nn.Block):
    def __init__(self, **kwargs):
        # 调用父类构造函数
        super(MLP, self).__init__(**kwargs)
        # 初始化隐藏层和输出层的参数
        with self.name_scope():
            self.dense0 = nn.Dense(256)
            self.dense1 = nn.Dense(10)

    def forward(self, x):
        return self.dense1(
            nd.relu(self.dense0(x)) # ReLU 放在这里噜
        )

    nn.Block Gluon 中的一个一般化的构件，你可以无限嵌套 Block 以构造新的 Block（可以是单个 Layer，也可以是一整个神经网络模型）。

    nn.Block 主要承担了以下任务：

    • 存储（部分）参数
    • 描述（部分）前向传递逻辑
    • 自动求导

    关于nn.Sequential

    • 实际上是 nn.Block 的容器，需要通过 add 添加 nn.Block

    • 会自动生成 forward 函数（就是串行执行被添加的 Block）

    参数初始化

    • 默认的 initialize() 函数会将参数初始化为 [-0.07, 0.07] 范围内均匀分布的随机数

    你也可以使用选项将他们初始化为一些怪东西：

    • 将参数初始化为符合 \(\mu=0,\sigma=0.02\) 的正态分布随机数：

      params.initialize(
    init=init.Normal(sigma=0.02), force_reinit=True
)

    • 初始化为全 1

      params.initialize(
    init=init.One(), force_reinit=True
)

    CNN 卷积神经网络

    Flatten 操作 => 卷积救一下啊！！！

    1. 会破坏像素中包含的语义
    2. 全连接导致输入过大

    1. 卷积层

      • 卷积核 kernel / filter：一个比较小的滑动窗口
      • 偏移量 bias：类似于 \(wX + b\) 的操作

      对于多通道输入，一次卷积后的结果为： $$ conv(data, w, b)[:,i,:,:] = conv(data, w[i,:,:,:]m,b[i]) $$

      其中 \(w[i]\) 为各通道的权重（卷积核），\(b[i]\) 为该通道对应的偏移量

    2. 池化层 Pooling

      进一步减少 feature 的数量，降低微小变化对模型的干扰

      • 最大池化：选用一个窗口中的 max 值
      • 平均池化：选用一个窗口中的 avg 值

    3. 全连接层

      会对输入做 Flatten 操作

    构造卷积神经网络
    weight_scale = 0.01
n_class = 10
n_fc = 128

# 1 - 卷积层1： 使用 5x5 kernel，输出 20 个通道
W1 = nd.random_normal(
    shape=(20,1,5,5), scale=weight_scale
)
b1 = nd.zeros(W1.shape[0])

# 2 - 卷积层2： 使用 3x3 kernel，输出 50 个通道
W2 = nd.random_normal(
    shape=(50,20,3,3), scale=weight_scale
)
b2 = nd.zeros(W2.shape[0])

# 3 - 全连接层1：包含 128 个节点
W3 = nd.random_normal(
    shape=(1250, 128), scale=weight_scale
)
b3 = nd.zeros(W3.shape[0])

# 4 - 全连接层2（输出层）：包含 10 个节点
W4 = nd.random_normal(
    shape=(W3.shape[1], 10), scale=weight_scale
)
b4 = nd.zeros(W4.shape[1])

params = [W1, b1 ~ W4, b4]
for p in params:
    p.attach_grad()

# 定义模型
def model(X):
    # 第一层卷积
    conv_1 = nd.Convolution(
        data=X,
        weight=W1, bias=b1,
        kernel=W1.shape[2:], num_filter=W1.shape[0]
    )
    act_1 = nd.relu(conv_1)
    pool_1 = nd.Pooling(
        data=act_1,
        pool_type="max",
        kernel=(2,2), stride=(2,2)
    )
    # 第二层卷积
    conv_2 = = nd.Convolution(
        data=pool_1,
        weight=W2, bias=b2,
        kernel=W2.shape[2:], num_filter=W2.shape[0]
    )
    act_2 = nd.relu(conv_2)
    pool_2 = nd.Pooling(
        data=act_2,
        pool_type="max",
        kernel=(5,5), stride=(2,2)
    )
    # 第一层全连接（线性回归）
    linear_1 = nd.dot(nd.flatten(pool_2), W3) + b3
    act_3 = nd.relu(linear_1)
    # 第二层全连接
    linear_2 = nd.dot(act_3, W4) + b4

    return linear_2

    Dropout 丢弃法解决过拟合

    用于多层神经网络

    DropOut 就是对 输入层/隐藏层 做以下操作：

    1. 随机选择该层的 部分输出
    2. 将这些元素置 0
    3. 对 非丢弃元素 进行 Flatten 操作
    DropOout 实现
    # drop_probability 定义了单个元素被丢弃的概率

def dropout(x, drop_probability):
    keep_probability = 1 - drop_probability
    assert 0 <= keep_probability <= 1
    # a) 全部丢弃
    if keep_probability == 0:
        return x.zeros_like()
    # b) 随机丢弃部分元素
    mask = nd.random.uniform(0, 1.0, x.shape) < keep_probability
    scale = 1 / keep_probability # 令 E(dropout(x)) == x
    return mask * x * scale
    • DropOut 的本质是：每次只激活全连接层中的部分节点
    • 我们可以为每一层单独设置 drop_probability，一般越靠前的丢弃概率越低

    VGG 复用构件

    手动定义 100 层的 AlexNet 将会十分抽象

    VGG 的特点是：使用一个较小的 kernel (3x3) 卷多次 + 一次池化，然后将这个小模块重复很多次

    实现 VGG 构件
    def vgg_block(n_convs, channels):
    out = nn.Sequential()

    with out.name_scope():
        for _ in range(n_convs):
            out.add(nn.Conv2D(
                channels=channels,
                kernel_size=3,
                padding=1, activation='relu'
            ))
        out.add(nn.MaxPool2D(
            pool_size=2, strides=2
        ))

    return out
    堆叠
    # 使用 architecture 定义堆叠样式
def vgg_stack(architecture):
    out = nn.Sequential()

    with out.name_scope():
        for (n_convs, channels) in architecture:
            out.add(vgg_block(n_convs, channels))

    return out

# 你可以在 VGG 结构后进行 flatten，然后塞全连接层

def VGG11():
    arch = ((1,64), (1,128), (2,256), (2,512), (2,512))
    net = nn.Sequential()
    with net.name_scope():
        # 先来一堆 VGG 堆叠
        net.add(vgg_stack(arch))
        # 摊平，开始全连接
        net.add(nn.Flatten)
        net.add(nn.Dense(4096, activation="relu"))
        net.add(nn.Dropout(0.5))
        net.add(nn.Dense(4096, activation="relu"))
        net.add(nn.Dropout(0.5))
        # 输出
        net.add(nn.Dense(n_output))

    BatchNorm 批量归一化

    虽然最终参数不一定更好，但是收敛更快

    • 我们通常会将 原始输入 的每一个 feature 都进行归一化 => 变成 \(avg = 0, \sigma = 1\) 的格式

    • 但在深度模型中，乘法效应 可能使得不同层的输入数据范围相差极大=> 采用相同的学习率会导致不同层的梯度下降效率极不一致

      于是我们：对 每层的输入 都进行归一化

    考虑一个小批次 \(B = (x_1, ... , x_m)\)：

    • 需要学习：拉升参数 \(\gamma\)，偏移参数 \(\beta\)
    • 归一化结果为： \(y_i = BN_{\gamma,\beta}(x_i)\)

    此外，定义：

    \[ \begin{align*} \mu_B \leftarrow \frac{1}{m} \sum x_i \tag{均值}\\ \sigma_B^2 \leftarrow \frac{1}{m}\sum(x_i - \mu_B)^2 \tag{方差}\\ \hat{x_i} \leftarrow \frac{x_i - \mu_B}{\sqrt{\sigma_B^2 + \varepsilon}} \tag{归一化的 X}\\ y_i \leftarrow y \hat{x_i}+\beta = BN_{\gamma,\beta}(x_i) \end{align*} \]

    此外，在实现时需要区分全连接层和卷积层：

    • 全连接层：对 每一个 batch 进行归一化
    • 而为卷积层：对每一个 channel 进行归一化，并保持四维形状
    def batch_norm(x, gamma, beta, eps=1e-5):
    assert len(x.shape) in (2,4)

    if len(x.shape) == 2: # 全连接层: 对每个 feature 求均值
        mean = x.mean(axis=0)
        variance = ((x - mean)**2).mean(axis=0)
    else:                 # 二维卷积: 对每个 channel 求均值
        mean = x.mean(axis=(0,2,3), keepdims=True)
        variance = ((x - mean)**2).mean(axis=(0,2,3), keepdims=True)

    # 均一化
    x_hat = (x - means) / nd.sqrt(variance + eps)

    # 拉升与偏移
    return gamma.reshape(mean.shape) * x_hat + beta.reshape(mean.shape)

    NIN

    Network In Network

    当我们把 Dense 层和 Conv 层进行混编时，会出现一个难题：

    • Dense 层需要 2D 的输入输出，而 Conv 则要求 4D
    • 直接 \(4D \rightarrow 2D\) 产生的全连接层有过多的参数

    对此，NIN 提出使用 1x1 的 kernel 实现：

    • 只对 channel 做全连接
    • 像素之间共享权重
    # 我们使用 1 * 正常 Conv + 2 * (1x1) Conv 来实现模块
# 后面两个 Conv 层充当了两个全连接层

def mlpconv(channels,
    kernel_size, padding, strides=1,
    max_pooling=True):
    out = nn.Sequential()
    with out.name_scope():
        out.add(
            # 正常全连接
            nn.conv2D(
                channels=channels, kernal_siez=kernal_size,
                strides=strides, padding=padding,
                activation='relu'
            )
            # 两个 1x1
            nn.conv2D(
                channels=channels, kernal_siez=1,
                strides=1, padding=0,
                activation='relu'
            )
            nn.conv2D(
                channels=channels, kernal_siez=1,
                strides=1, padding=0,
                activation='relu'
            )
        )
        if max_pooling:
            out.add(nn.MaxPool2D(
                poolsize=3, strides=2
            ))
    return out

    NIN 实际上取消了全连接层，最后使用平均池化 + Flatten 直接得到标量输出
