    Axios 的使用

    发送网络请求

    0 安装依赖

    npm install axios

    1 引入并使用

    // @ component.jsx
import axios from 'axois'
// 一个 GET 请求的 Sample
axois.get('URL').then({
    res => {console.log(res.data)},  // 成功的回调
    err => {console.log(err.message)}// 失败的回调
})

    代理的配置参考这里

    其他基本使用方式

    1 - 使用 axios 本体进行请求

    GET(ReadOnly)
    axios({
    method: 'GET',
    // 是 Json-Server 的 url 地址
    url: 'http://localhost:3000/posts/1'
}).then( res => {} )
    POST(Insert)
    axios({
    method: 'POST',
    url: 'http://localhost:3000/posts',
    data: {
        title: 'This is title',
        author: 'Tom'
    }
}).then( res => {} )
    PUT(Update)
    axios({
    method: 'PUT',
    // 2 - 需要修改记录的 id
    url: 'http://localhost:3000/posts/2',
    data: {
        title: 'New World',
        author: 'Jack'
    }
}).then( res => console.log(res))
    DELETE
    axios({
    method: 'DELETE',
    // 1 - 需要删除记录的 id
    url: 'http://localhost:3000/posts/1'
}).then(res=>{})

    2 - 使用本体上的一些函数

    axios.request
    axios.request({
    method: 'GET',
    url: 'http://localhost:3000/cpmments',
}).then(res => {})
    axios.post
    axios.post(
    'url',
    // data
    {
        'body': 'sth',
        'postId': 2
    }
).then(res=>{})

    响应结构

    • config 配置对象
    • data 服务器返回的结果
    • headers 响应头的信息
    • request 原生 ajax 实例对象
    • status 响应状态吗
    • statusText 响应状态字符串

    使用配置对象

    1 - 进行全局配置

    进行一些通用配置项的书写 

    // 进行了一个全局的配置
axios.defaults.baseURL == 'http://localhost:3000'
// 这样写之后的默认方式就是GET（可以不写了），其他方法就独立配置盖过去
axios.defaults.method = 'GET'
// 3s后无响应就取消请求
axios.defaults.timeout = 3000
// 其他配置项也都可以设置成 defaults 的！

    2 - 对实例进行配置

    const a = axios.create({
  // 此处传入的 config 只对实例对象 a 起作用
    baseURL: 'http://localhost:3000',
    timeout: 2000
})
// 然后就可以直接用实例对象去发请求（用起来和直接抓 axios 本体差不多）
a({url:'/comments'}).then(res=>{})
a.get('/getSth').then(res=>{})

// 在对不同服务器发请求时，可以使用不同的实例（少写一些 baseURL）

    取消请求

    var cancel = null; // 必须是全局变量
// 首先要为请求对象配置 cancelToken
axios({
    method: 'GET',
    url: 'http://localhost:3000/posts',
    cancelToken: new axios.CancelToken(function(c) {
        cancel = c;
    })
}).then(res=>{})

// 在绑定了取消事件的按钮里执行下面这个函数就行
cancel()

    然后实现：在上一个请求还没有返回前，取消前一个请求（？） 

    var cancel = null;
// cancel 不为 null 时，说明上一次请求
if(cancel !== null) cancel();
axios({
    methods: 'GET',
    url: 'http://localhost:3000/comments',
    cancelToken: new axios.CancelToken(c => cancel=c)
}).then(
     res => {
         cancel = null;
     }
)

    使用 mkdocs 快速搭建静态站点

    需要预先安装 python

    1 虚拟环境

    # 创建虚拟环境
python -m venv [venv_name]
# 激活虚拟环境
./venv_name/Scripts/activate # 运行脚本（大概）- 命令行开头出现(venv)就算成功

# 推出虚拟环境
deactivate

    windows 下激活环境的命令稍微有点不同，搜 venv 就会有相关教程

    2 使用 mkdoc

    # 1 安装 mkdocs
pip install mkdocs # 不行的话用 pip3 install mkdocs
# 2 检查是否正常安装
mkdocs --version # 出现相关信息就ok
# 3 创建项目
mkdocs new [prj_name]
# 4 开启服务
cd prj_name
mkdocs serve # 正常运行后 bash 会出现本地地址，访问就行

    配置多级目录可以自己搜一下 mkdocs 配置。

    3 使用 material 主题

    # 安装依赖
pip install mkdocs-material
# 更改配置 mkdocs.yml，添加
theme: name: material
# 重新 serve 一遍应该就行

    4 发布

    mkdocs build
# 打包生成 /site 目录，把里面的东西全丢到 github 仓库根路径下就行

    另外要给 Github 仓库配置 Git Page，教程可以看这里

    然后就可以通过 URL 访问你的静态站点哩！
