    3 数据处理

    1 字符串

    1.1 常用函数

    一些来自于 strings 包的常用函数介绍，这些操作均 严格区分大小写

    • Contains：返回 bool 判断是否包含制定子串

      func Contains(s, substr string) bool

Println(strings.Contains("seafood", "sea")) // 返回 true

    • Join：将多个字符串通过指定串连接

      func Join(a []string, sep string) string

s := []string{"str1", "str2", "str3"}
Println(strings.Join(s, "-")) // 返回 "str1-str2-str3"

    • Index：查找字串的起始位置，找不到返回 -1

      func Index(s, sep string) int

Print(strings.Index("seafood", "ood")) // 返回 4
Print(strings.Index("seafooa", "tea")) // 返回 -1

    • Repeat：返回指定字符串重复 n 次的结果

      func Repeat(s string, count int) string

Print("ba" + string.Repeat("na", 2)) // 返回 "banana"

    • Replace：将 old 子串替换为 new 子串，count 指定替换前 N 个（小于 0 全部替换）

      func Replace(s, old, new string, count int) string

strings.Replace("ba ba ba", "ba", "na", 2)  // "na na ba"
strings.Replace("ba ba ba", "ba", "na", -1) // "na na na"

    • Split：按照指定子串 sep 分割，返回 slice

      func Split(s, sep string) []string

strings.Split("a,b,c", ",") // ["a", "b", "c"]
strings.Split("abc", "")    // ["a", "b", "c"] 全分割

    • Trim：掐头去尾制定子串

      func Trim(s string, cutset string) string

// 去头尾空格
string.Trim("   Achtung ", " ") // 返回 "Achtung"

    • Fileds：去掉所有空格，并按照空格划分 token

      func Fields(s string) []string

string.Fields("  I am the strom ")
// 返回 ["I", "am", "the", "storm"]

    1.2 字符串转换

    来自 strconv 包

    • Append 系列：将其他类型转 string，并追加至当前 []byte

      str := make([]byte, 0, 100)
str = strconv.AppendInt(str, 4567, 10) // 以 Decimal 方式追加
str = strconv.AppendBool(str, false)
str = strconv.AppendQuote(str, "abc")
str = strconv.AppendQuoteRune(str, "啊哈")

    • Format 系列：将其他类型转 string

      var str string
str = strconv.FormatBool(true)
// f 指以小数方式打印，-1 指小数点位数，64 表示处理为 float64
str = strconv.FormatFloat(3.14, 'f', -1, 64)
str = strconv.Itoa(666)

    • Parse 系列：将 string 转成其他类型

      // string 转 bool
var flag bool
var err error
flag, err = strconv.ParseBool("false")
if err != nil {
    Print(err)
}

// string 转 int
var i int
i, err = strconv.Atoi("75")

    1.3 正则表达式

    能用 strings 解决的就不要用正则

    死去的编译原理又开始攻击我

    1.3.1 正则语法

    Go 实现的是 RE2 标准的正则语法，默认采用 perl 标志，部分可以关闭

    RegExp Desc.
    . 除换行符（ s==true 时包含换行符）
    [xyz] / [x-z] OR
    [^xyz] NOT
    \d 数字
    \D Not(数字)
    [:alpha:] ASCII 字符集合
    [:^alpha:] Not(ASCII 字符集合)
    \pN Unicode 字符集合（单字符）
    \PN Not(Unicode 字符集合 S)
    \p{Greek} Unicode 字符集合（完整）
    \P{Greek} Not(Unicode 字符集合 C)

    1.3.2 实用

    import "regexp"

// 1. 定义规则
reg := regexp.MustComile('a.c') // "a?c"
if reg == nil {
    Print("err = ", reg)        // 正则 bug
}

// 2. 解析字符串
str = "abc adc bac bdc cad"
res := reg.FindALlStringSubmatch(str, -1)

    2 JSON

    JSON 是好文明！

    2.1 JSON 字符串生成

    通过包 encoding/json 实现

    • 通过 struct 生成 JSON 字符串

      import "encoding/json"

// 定义待转换的结构体（类型和字段首字母必须大写）
// 生成的字段名 == 成员变量名
type Customer struct {
    Name string     `json:"aha"`     // 强行改小写字段名
    Company string  `json:"-"`       // 不输出该字段
    id      int     `json:",string"` // 转 str 类型输出
    Order []string  // => Order: ["", "", ...]
}

// 定义结构体变量并初始化
var c Customer = Customer {
    "Sean",
    "ZJE",
    []string{"Coffer", "Latee"}
}

// 生成 JSON -> json.Marshal() 接受任何类型的参数
info, err := json.Marshal(c)
// 格式化：有换行和缩进
info, err := json.MarshalIndent(c, "", "  ")
if err != nil {
    Print(err)
}
// info 本身是 []byte，直接打印会出 ASCII
Print(string(info)) // 转 string 显示

    • 通过 map 生成 JSON

      // 创建 map -> 实用空接口兼容不同的 valType
m := make(map[string]interface{}, 4)
m["name"] = "Sean"
m["company"] = "ZJE"
m["order"] = []string{"Coffer", "Latee"}

// 生成 JSON
info, err := json.Marshal(m)

    2.2 JSON 解析

    • 解析到 struct

      // 待解析 JSON 字符串，需要转成 []byte
jsonStr := `
{
    "comp": "ZJUi",
    "operator": "Jake"
}`

// 用于接受的 struct -> 重写字段名进行匹配
type Person struct {
    Company string `json:"comp"`
    Name string    `json:"operator"`
}

var p Person // 用于接受的结构体变量，需要取地址（不然没法改）
err := json.Unmarshal([]byte(jsonStr), &p)
if err != nil {
    Print(err)
}

    • 解析到 map

      m := make(map[string]interface{})          // 用于收数据的 map
err := json.Unmarshal([]byte(jsonStr), &m) // 取地址
if err != nil {
    Print(err)
}

// 用类型断言取数据：没办法直接取啊喂！
var str string
for _, item := range m {
    switch data := item.(type) {
        case string:
            str = data
        case []interface{}:
            // 用来接 []string -> 还得重新断言（太抽象了）
    }
}

    一个综合案例：

    path = "./readme.txt"

func WriteFile(path string) {
    // 新建并打开
    fptr, err := os.Create(path)
    if err != nil {
        Print(err)
        return
    }

    // 使用后关闭文件
    defer fptr.Close()

    // write sth
    var buffer string
    for i := 0 ; i < 10 ; i++ {
        buffer = fmt.Sprintf("i = %d\n", i)
        _, err := fptr.WriteString(buffer)
        if err != nil {
            Print(err)
        }
    }
}

func ReadFile(path string) {
    // 打开文件
    fptr, err := os.Open(path)
    if err != nil {}

    // 使用后关闭
    defer fptr.Close()

    // read sth
    buffer := make([]byte, 1024*2)    // 每次读取一个 buffer 大小的数据
    len, err := fptr.Read(buffer)     // len 为获取信息总长
    if err != nil && err != io.EOF {} // 排除 EOF 的干扰
    Print(sting(buffer[:len]))
}

// 通过 bufio 包实现按行读取
func ReadByLine(path string) {
    // 打开关闭

    // 新建 Reader
    reader := bufio.NewReader(fptr)

    for {
        // 按行读取 -> 碰到 `\n` 就暂停（包含 `\n`）
        buf, err := reader.ReadBytes('\n')
        if err != nil {
            if err == io.EOF {
                break
            } else {
                Print(err)
            }
        }
        Printf("%s", string(buf))
    }
}

// 实现拷贝文件，从命令行参数获取 src dst
func main() {
    args := os.Args
    if len(args) == 3 {         // run 本身算一个参数
        if args[1] == args[2] { // src != dst
            return
        }
    }
    src, dst := args[1], args[2]

    // 以只读模式打开 src
    srcF, err1 := os.Open(src)
    if err!= nil return

    // 新建 dst
    dstF, err2 := os.Create(dst)
    if err2 != nil return

    // 延迟关闭
    defer srcF.Close()
    defer dstF.Close()

    // 开始 copy
    buf := make([]byte, 4*1024) // 4k 缓冲区
    for {
        len, err3 := srcF.Read(buf)
        if err3 != nil {
            if err3 == os.EOF break
        }
        // 读多少写多少
        dstF.Write(buf[:len])
    }
}
