    4 并发处理

    并行 & 并发
    • 并行 Parallel：同一时刻有多条指令在多个处理器上同时执行
    • 并发 Concurrent：多个进程快速轮换执行（宏观上，多个进程同时活跃）

    1 GoRoutine

    普通 PC 跑 10+ thread 就差不多寄了，但是跑 1w+ goroutine 轻轻松松

    • GoRoutine 是一个 “协程”（比 thread 粒度更小）
    • Go 本身支持 goroutine 之间进行 内存共享，因此只需要极少的栈内存

    1.1 创建

    • 我们使用 go 关键字定义并发执行单元 goroutine，调度器会自动将其安排至合适的 thread 上执行

    • 主函数 main() 运行在单独的 goroutine（ main goroutine ）中

    主协程结束会强制结束所有子协程

    func task() {
    for {
        Println("in task")
        time.Sleep(time.Second)
    }
}

func main() {
    go task()                   // 为 task() 单独创建 gouroutine

    for {                       // 在 main routine 中持续执行
        Println("in main")
        time.Sleep(time.Second) // 没有 “让出” 时间片，只是在磨洋工
        if condition return     // 满足条件时结束 main routine
    }

    // 如果将 go task() 放在此处，会因为 unreachable 而没有动静捏
}

    主协程直接退出会导致子协程来不及调用

    func main() {
    go func() {
        for {
            Println("in task")
            time.Sleep(time.Second)
        }
    }()
    // 不会有任何输出：还没 go 起来 main() 就已经结束了
    // 所以记得在 main() 里丢一个 dead loop
}

    1.2 Runtime

    来自 runtime 包的方法

    1.2.1 Gosched

    强制让出当前 routine 的时间片，下次从断点恢复执行

    // 一个错误案例
func main() {
    go func() {
        for i:=0 ; i<3 ; i++ {
            Print("task")
        }
    }()

    // 事实上输出俩 main 就结束了 -> 主协程没有让出时间片，自己跑完溜了
    for i:=0 ; i<2 ; i++ {
        // runtime.Gosched() // 正确做法：先让子协程跑 -> task*3 + main*2
        Print("main")
    }
}

    1.2.2 Goexit

    立即结束当前 routine（当然，先要执行本 routine 中的所有 defer ）

    func task() {
    defer Print('t-1')

    return            // 直接结束 task() -> 输出 f-1, t-1, f-2
    runtime.Goexit()  // 结束 routine   -> 输出 f-1, t-1（跑路了）

    Print('t-2')
}

func main() {
    go func() {
        Print('f-1')
        task()
        Print('f-2')
    }()

    for {} // 照顾 sub routine 的 dead loop
}

    1.2.3 GoMAXPROCS

    设置可用于并行计算的最大 CPU 核数，并返回 之前的值

    func main() {
    runtime.GoMAXPROCS(1) // 单核

    for {
        go Println(1)     // 狂开子协程（但没什么用）
        Println(0)
    }
}

    2 Channel

    用于实现对资源的互斥访问

    由于 goroutine 运行在相同的地址空间，因此必须做好进程间同步

    goroutine 通过通信以正常共享内存（而非通过共享内存来进行通信）

    2.1 Channel 类型

    与 map 类似，channel 也是一个对 make 创建底层数据结构的引用

    • 当我们复制一个 channel 实例，或用于参数传递时，我们只是对 channel 进行了值拷贝

      但由于拷贝的是一个 地址，因而可以操作同一份原始数据

    • channel 的零值是 nil，对其进行读写均会被阻塞

    • channel 的定义

      需要制定包含的值类型，我们可以通过 make() 函数生成 channel 实例

      make(chan Type)            // 等价于指定 make(chan Type, 0)
make(chan Type, capacity)
      • capacity == 0 的 channel 是阻塞读写的 => 不需要显示指定 lock
      • capacity > 0 时，会在写满 cap 个字符后阻塞写入进程（有缓冲）

    • 收发数据

      使用 <- 收发数据
      // 可以通过 len() 来查看当前通道中的数据总数

// 写入数据
channel <- value

// 读取数据
<- channel           // 读取并丢弃
x := <- channel      // 读取并赋值给 x
x, ok := <- channel  // 读取、赋值，同时检查 chan 是否关闭/为空

    2.1.1 无缓冲 channel（同步）

    capacity == 0 的 channel 即为无缓冲 channel，可以通过以下两种方式定义：

    ch := make(chan type)
ch := make(chan type, 0) // 显式指定容量为 0

    无缓冲通道对于 读写双方 都是阻塞的：

    • 若无人接收数据，写入进程将持续阻塞
    • 若无人写入数据，接收进程将持续阻塞

    区别于 cap == 1 的通道：写入进程可以在写入一个数据后直接离开

    2.1.2 缓冲 channel（异步）

    只要在创建时制定 > 0 的容量，即可创建有缓冲的通道

    • 写满后，阻塞写入进程
    • 读空后，阻塞接收进程

    2.1.3 单向 channel

    通道默认是双向的 => 这意味着我们能够自由的从通道中读写数据

    • 单向通道需要显示声明，格式如下：

      使用 <- & chan 进行组合定义
      var send chan<-  int // 仅写
var recv <-chan  int // 仅读

    • 支持将双向通道隐式转换为单向的 只读/只写 队列：

      但不允许将单项通道转换为双向通道

      ch := make(chan int)        // 创建双向通道

var writeCh chan<- int = ch // 隐式转换为仅写通道
writeCh <- 1                // 合法
<- writeCh                  // 寄！

var readCh  <-chan int = ch // 隐式转换为仅读通道

    2.2 实现同步

    2.2.1 顺序执行

    User1 执行完毕后，执行 User2
    // 创建全局 channel
var ch = make(chan int)

func main() {
    go User1()
    go User2()
    for {}
}

func User1() {
    Printer("Hello")
    ch <- 1           // 向通道写入数据表示操作完毕
}

func User2() {
    <- ch            // 只要通道中没有数据，本操作就会阻塞
    Printer("World")
}

func Printer(str string) {
    for _, char := range str {
        Printf("%c", char)
        time.Sleep(tiem.Second)
    }
    Printf("\n");
}

    2.2.2 子协程结束后再终止主程

    func main() {
    ch := make(chan int)

    // 启动子协程
    go func() {
        defer Println("子协程结束")
        for i:=0 ; i<5; i++ {
            Println("i = ", i)
            time.Sleep(time.Second)
        }
        ch <- 1
    }()

    <- ch // 阻塞，直至子协程执行完毕
}

    2.2.3 关闭通道同时终止两个进程

    我们可以使用 close(channel) 来关闭指定通道，通道关闭后：

    • 将无法继续写入数据，否则将引发 panic 并立即返回零值
    • 可以继续读取数据
    func main(){
    ch := make(chan int)

    go func(){
        for i:=0 ; i<5 i++ {
            ch <- i
        }
        // 收摊后，关闭通道
        close(ch)
    }()

    for {
        if num, ok := <- ch ; ok == true {
            Println("i = ", num) // 管道未关闭，打印结果
        } else {
            break;               // 管道关闭，跳出死循环以终止进程
        }
    }

    // 通过 range 遍历可以自动在关闭后跳出循环
    for data := range ch {
        Println("i = ", data)
    }
}

    2.2.4 单向通道解决 生产-消费 问题

    实际上是将同一个双向通道分别隐式映射为不同类型的单向通道，从而限制 生产者/消费者 的行为

    func main() {
    ch := make(chan int) // 此处定义的是双向通道

    go Producer(ch)

    Consumer(ch)         // 主程将在 Consumer() 执行完毕后结束
}

// 通过参数类型直接隐式转换为单向通道
func Producer(out chan<- int) {
    for i:=0 ; i<10 ; i++ {
        chan <- i*i
    }
    close(ch)
}
func Consumer(in <-chan int) {
    for num := range in {
        Println("square = ", num)
    }
}

    3 定时器

    3.1 Timer

    setTimeout()，仅响应一次

    没错就是定时器，你可以告诉 timer 需要在多长的时间间隔后执行该操作

    • Timer 为我们提供了一个 channel，该通道将在触发时提供 触发时刻时间值：

      验证 channel 内容
      func main() {
    // 创建一个 delay = 2s 的 timer
    timer := time.NewTimer(2 * time.Second) // 此时为 t 时刻

    // 2s 后触发 timer 事件，timer 向 channel 写入即时时间（t+2）
    data := <-timer.C    // timer.C 是该定时器提供的通道（未触发前，读操作阻塞）
    Print(data)           // data = t+2
}

    • 停止 / 重置 Timer：

      func main(){
    timer := time.NewTimer(3 * time.Second)
    timer.Reset(1 * time.Second) // 立即将 period 重置为 1s（返回 bool 判读是否成功）

    go func(){
        <-timer.C
        Println("Begin Subroutine")
    }()

    timer.Stop() // 停止 Timer
}
    使用 timer 实现延时功能
    // 个人评价是不如 JS -> 不能同时指定 period & callback
func main() {
    <-time.After(2 * time.Second) // 返回的是 channel，本质是在这一句阻塞 2s
    Println("After 2s")
}

    事实上 NewTimer() & Sleep() & After() 可以实现完全一致的功能（？）

    3.2 Ticker

    setInterval()，每隔指定周期响应一次

    本质是每隔 T 向 channel 中写入数据
    func main() {
    ticker := time.NewTicker(time.Second)

    i := 0
    for {
        <-ticker.C // 通过读取操作实现固定周期阻塞
        i++
        Println("i = ", i)
        // 在执行给定次数后关闭 ticker
        if i == 5 {
            ticker.Stop()
            break
        }
    }
}

    4 Select 关键字

    我们可以通过 select 关键字监听 channel 上的数据流动

    select 的整体语法与 switch 相似，但每个 case 必须是一个 IO 操作

    // 若存在多个可执行 case，则会随意挑选其中一个进行执行  
select {
    case <- chan1:
        // 若能从 chan1 中读取数据，执行该分支
    case chan2 <- 1:
        // 若能向 chan2 中写入数据，执行该分支
    default:
        // 若不存在 default 则阻塞至任一 case 能成功执行
}
    select 实现 Fibonacci
    func main() {
    ch := make(chan int)
    quit := make(chan bool)

    // 消费者协程（发出停止信号，控制 Fib[n]）-> 需要先 go 起来
    go func(n int) {
        for i := 0 ; i < n ; i++ {
            num := <- ch
            Println("Fib[", i ,"] = ", num)
        }
        quit <- true // 发出停止信号
    }(5)

    fib(ch, quit) // 生产者协程 -> 先执行 fib 会因为 select 无法执行而持续阻塞
}

func fib(out chan<- int, quit <-chan bool) {
    x, y := 1, 1
    for {
        // 监听 channel 中的数据
        select {
            case out <- x: x, y = y, x+y // 很妙的递推式实现
            case <- quit: return
        }
    }
}

    超时机制实现

    通过 select 避免程序中所有 goroutine 同时阻塞

    // 在读取 0~4 后的 3s 退出整个程序
func main() {
    ch := make(chan int)
    quit := make(chan bool)

    go func() {
        for {
            select {
                case num := <-ch:
                    Println("recv: ", num)
                case <-time.After(3 * time.Second):
                    Println("超时...")
                    quit <- true
            }
        }
    }()

    for i:= 0 ; i < 5 ; i++ {
        ch <- i
    }

    <- quit
    Println("程序结束")
}
