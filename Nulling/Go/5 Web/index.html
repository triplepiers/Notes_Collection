    5 网络编程

    网络通信条件

    • 物理地址：网卡 MAC地址，可以通过 ARP 协议从 IP 匹配
    • 逻辑地址：IP 地址，用于确定接收主机
    • 端口号：用于确定主机上的具体进程

    1 Socket

    套接字的本质是两个单向 channel，适用于 C/S 模式

    常用的 Socket 有两类：

    1. 流式 SOCK_STREAM：面向连接，针对 TCP
    2. 数据报式 SOCK_DGRAM：无连接，针对 UDP

    1.1 TCP 并发服务器

    服务器需要持续监听用户请求

    处理单个用户请求
    import "net"

// 处理用户请求
func Handle(conn net.Conn) {
    defer conn.Close() // 本函数结束后关闭连接

    addr := conn.RemoteAddr().String()
    // 获取用户地址
    Println("Client from ", addr)

    for {
        // 读取用户数据
        buf := make([]byte, 1024) // 1024B 的缓冲切片
        len, err2 := conn.Read(buf)
        if err2!= nil {
            Println("读取失败，err = ", err2)
            continue
        }

        // 打印用户数据（收多少打多少）
        Printf("[%s]: %s\n", addr, string(buf[:len]))

        // 检测关闭请求 ?msg == "exit"，实际发送的是 exit\r\n
        if string(buf[:len-2]) == "exit" {
            Printf("Client [%s] exits\n", addr)
            return
        }

        // 返回 Upper(msg)
        conn.Write([]byte(strings.ToUpper(string(bud[:len]))))
    }
}

func main() {
    // 监听：指定协议、IP、Port
    lietener, err := net.Listen("tcp", "127.0.0.1:8080")
    if err!= nil {
        Println("监听失败，err = ", err)
        return
    }
    defer listener.Close()

    // 阻塞，直至有用户发起请求
    for {
        conn, err1 := listener.Accept
        if err1!=nil {
            Println("连接失败，err = ", err1)
            continue
        }

        // 新建协程接收用户请求
        go Handle(conn)
    }
}

    1.3 文件传输

    整个过程分为下述两个阶段：

    1. 收发文件名
    2. 收发文件内容
    Sender
    func main() {
    // 从命令行参数获取文件名
    list := os.Args
    path := list[1]

    // 获取文件信息 -> info.Name()/Size() 分别是文件名/大小
    info, err := os.Stat(path)
    if err!=nil {
        Println("fail to open file: ", err)
        return
    }

    // 主动连接服务器
    conn, err1 := net.Dial("tcp", ":8080")
    if err1 := nil {}
    defer conn.Close()

    // 发送文件名
    conn.Write([]byte(info.Name()))

    // 接收回复（如果是 "ok" 继续发送）
    buf := make([]byte, 1024)
    len, err2 := conn.Read(buf)
    if buf[:len] == "ok" {
        // 发动文件内容
        SendFile(path, conn)
    }
}

// 用于发送文件内容的函数
func SendFile(path string, conn net.Conn) {
    // 只读方式打开文件
    f, err := os.Open(path)

            }
        }

}

    1.4 并发聊天室

    Server 需要：

    • 新用户上线 / 用户发送信息时，进行全局广播（监听 chan broadcast 实现）

      信息由 chan broadcast 再分发至单独的 client.C 实现单播

    • 维护用户信息 Map 以确定是否达到并发上限 / 群发信息，Map 与节点结构如下：

      type Client struct {
    C chan string;   // 管道：接收用户信息
    addr   string;
}
    Server
    var userMap map[string]Client   // 全局 Map: addr -> Client_node
var broadcast := make(chan string) // 全局广播通道

func main() {
    // 1. 注册监听器
    listener := net.Listen("tcp", ":8080")
    defer listener.Close()

    // 2. 用于群发的协程（放在 dead-loop 后面就执行不到了）
    go Broadcaster()

    // 3. 循环阻塞，监听用户连接（主协程）
    for {
        conn := listener.Accept()

        go Handle(conn)      // 3. 为每个用户单独注册协程处理通信
    }
}

// 广播捏
func Broadcaster() {
    // 给 userMap 分配内存空间
    userMap = make(map[string]Client) 

    for {  // 持续从广播信道中读取信息
        msg := <-broadcast  
        for _, client := range userMap { 
            client.C <-msg // 遍历以群发消息（丢给 channel，具体由对应协程处理）
        }
    }
}

// 用户接待函数
func Handle(conn net.Conn) {
    defer conn.Close()

    // 将新上线用户加入 userMap
    addr := conn.RemoteAddr().String()
    client := Client(make(chan string), addr)
    userMap[addr] = client

    // Server -> Client 协程
    go SendMsg(client, conn)

    // 广播上线信息
    broadcast <- "Client @" + addr ": login"

    isQuit := make(chan bool)        // 判断用户是否主动退出
    hasData := make(chan bool)       // 用于超时检测
    // 接收用户信息的协程
    go func() {
        buf := make([]byte, 1024)
        for {
            len, err := conn.Read(buf)
            if err!=nil {            // 出错或断开连接
                isQuit <- true
                return
            }   
            // 全局转发
            msg := string(buf[:len])
            broadcast <- "Client @" + addr + ": " + msg
            hasData <- true
        }
    }

    for {
        select {
            case <- isQuit:  // 注销 Map 并广播信息
                delete(userMap, addr)
                message <- "Client @" + addr + ": logout"
            case <- hasData: // 超时检测
            case <- tiem.After(60 * time.Second):
                delet(userMap, addr)
                message <- "Client @" + addr + ": timeout"
                return
        }
    }
}

// 单发消息
func SendMsg(client Client, conn net.Conn) {
    for msg := range client.C { // 持续监听 client.C 中的数据
        conn.Write([]byte(msg + "\n"))
    }
}

    2 HTTP

    适用于 B/S 模式，使用 net/http 包自动形成报文

    2.1 简单服务器

    import "net/http"

func main() {
    // 路由 -> 自动调用指定回调
    http.HandleFunc("/", handleIndex)

    // 监听指定端口
    http.ListenandServe(":8080", nil)
}

// 处理主页访问请求 -> 回复 hello world
func handleIndex(w http.ResponseWriter, req *http.Request) {
    w.Write([]byte("hello world!"))
    // 我们可以从 Request 结构体中获取 Method，Form 参数等信息
}

    2.3 爬虫

    编写爬虫的基本流程如下：

    1. 明确目标
    2. 把所有 HTML 爬下来
    3. 丢掉没有用的部分（还是得分析 tag 捏）
    4. 保存需要的部分
    简单并发
    // 单页面处理协程
func GetPage(i int, page chan<- int) {
    // 构建本页 URL
    url := baseURL + strconv.itoa((i-1)*50) 

    // 爬取指定页面
    res, err := Get(url)
    if err!=nil {
        Println(err)
        return
    }

    // 提取关键信息（正则立大功）这边尝试筛选 h1 标签里面套的超链接
    reg := regexp.MustCompile(`<h1 class="dp-b"><a href="(?s:(.*?))"`)
    urls := reg.FindAllStringSubmatch(res. -1) // -1: 匹配所有字串


    // 将内容写入本地文件（以页码作为文件名）
    fileName := strconv.itoa(i) + ".html"
    f, err1 := os.Create(fileName)
    if err1!=nil { return }
    for _, url := range urls {
        f.Write(url + "\n")
    }
    f.Close()

    // 向通道写入已完成页码
    page <- i
}

// 爬取并处理指定页面
func Get(url string)(res string, err error) {
    response, err1 := http.Get(url)
    if err1!=nil {
        err = err1
        return
    }
    defer response.Body.Close()

    // 读取网页内容
    buf := make([]byte, 1024*4)
    for {
        len, err := response.Body.Read(buf)
        if len==0 {
            break;
        }
        res += string(buf[:len])
    }

    return
}

func main() {
    var start, end int         // 输入起始/终止页码
    page := make(chan int)  // 记录页码，延迟关闭

    // 贴吧每翻一页 pn+=50
    for i:=start ; i<=end ; i++ {
        go GetPage(i, page)
    }

    // 循环 n 次，保证所有页面完成
    for i:=start ; i<end ; i++ {
        Printf("Page %d finished\n", <-page)
    }
}
